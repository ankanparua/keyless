resource "aws_iam_role_policy" "ci_cd_policy" {
  name = "CI-CD-S3-Read"
  role = aws_iam_role.ci_cd_role.id

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
# Permission to list the names of ALL buckets in the account
#=================New-addition=================
        Effect   = "Allow"
        Action   = "s3:ListAllMyBuckets"
        Resource = "*" 
      },
      {
        # Permission to look inside and read from your specific bucket
        Effect   = "Allow"
        Action   = [
          "s3:GetObject",
          "s3:ListBucket"
        ]
        Resource = [
          "arn:aws:s3:::${var.github_org}-deployment-bucket",
          "arn:aws:s3:::${var.github_org}-deployment-bucket/*"
        ]
      }
    ]
  })
}
